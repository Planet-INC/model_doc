We use the following notations:
\begin{notation}
\divergex{x}{y} = \sum_{i}\doverd{y_i}{x}
\end{notation}
\begin{notation}
\vectProd{x}{y} = \sum_{i}y_i x_i
\end{notation}
\begin{notation}
\gradien{x}{y} = \vect{\left(\doverd{y}{x}\right)}
\end{notation}
\begin{notation}
\diverge{F} = \text{div}(\vect{F}) = \doverd{F_x}{x} + \doverd{F_y}{y} + \doverd{F_z}{z}
\end{notation}
